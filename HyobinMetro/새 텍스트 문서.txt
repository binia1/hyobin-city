        // 1. 역 리스트 렌더링
        function renderStations(data) {
            const grid = document.getElementById('stationGrid');
            grid.innerHTML = '';
            data.forEach(st => {
                const card = document.createElement('div');
                card.className = 'station-card bg-white rounded-xl border border-slate-200 overflow-hidden cursor-pointer p-5 group';
                card.onclick = () => openModal(st);
                card.innerHTML = `
                    <div class="flex justify-between mb-2">
                        <span class="${st.colorClass} text-xs font-bold px-2 py-1 rounded">${st.lineName}</span>
                        <span class="text-slate-400 text-xs">${st.code}</span>
                    </div>
                    <h3 class="text-lg font-bold text-slate-800 mb-1 group-hover:text-sky-600">${st.name}</h3>
                    <p class="text-xs text-slate-500 truncate">${st.address}</p>
                `;
                grid.appendChild(card);
            });
        }

        // 2. 검색
        function filterStations() {
            const query = document.getElementById('searchInput').value.toLowerCase();
            renderStations(stations.filter(st => st.name.includes(query)));
        }

        // 3. ★ 핵심: 워드 파일 뷰어 실행 함수
        async function openModal(st) {
            const modal = document.getElementById('stationModal');
            const header = document.getElementById('modalHeader');
            const container = document.getElementById('docx-container');
            const spinner = document.getElementById('loadingSpinner');
            const downloadLink = document.getElementById('downloadLink');

            // UI 초기화
            header.className = `p-5 text-white flex justify-between items-center ${st.colorClass}`;
            document.getElementById('modalLineBadge').innerText = st.lineName;
            document.getElementById('modalName').innerText = st.name + '역';
            downloadLink.href = st.fileUrl || '#';
            
            // 뷰어 컨테이너 비우기 (중요: 이전 내용 삭제)
            // 단, 스피너는 지우지 않기 위해 while문 사용
            while (container.firstChild) {
                if (container.firstChild.id === 'loadingSpinner') break;
                container.removeChild(container.firstChild);
            }

            modal.classList.remove('hidden');
            document.body.style.overflow = 'hidden';

            // 파일이 없으면 종료
            if (!st.fileUrl) {
                container.innerHTML += '<div class="flex items-center justify-center h-full text-slate-400">등록된 문서가 없습니다.</div>';
                return;
            }

            // 워드 파일 불러오기 & 렌더링
            try {
                spinner.classList.remove('hidden'); // 로딩 시작

                const response = await fetch(st.fileUrl);
                if (!response.ok) throw new Error("파일을 찾을 수 없습니다.");
                
                const blob = await response.blob();
                
                // docx-preview 라이브러리 사용
                await updateAsync(blob, container, container.firstChild); // 컨테이너에 렌더링

            } catch (error) {
                console.error(error);
                container.innerHTML += `<div class="flex flex-col items-center justify-center h-full text-red-500 font-bold"><i class="fa-solid fa-circle-exclamation text-4xl mb-2"></i>문서를 불러오지 못했습니다.<br><span class="text-xs font-normal text-slate-500">Live Server로 실행했는지 확인해주세요.</span></div>`;
            } finally {
                spinner.classList.add('hidden'); // 로딩 끝
            }
        }

        // docx-preview 렌더링 옵션
        function updateAsync(data, target, before) {
            return docx.renderAsync(data, target, target, {
                className: "docx-wrapper", // 스타일 클래스
                inWrapper: true,
                ignoreWidth: false,
                ignoreHeight: false,
                breakPages: true, // 페이지 나누기
                experimental: false
            });
        }

        function closeModal(e) {
            if (e && e.target !== e.currentTarget && e.target.tagName !== 'BUTTON' && !e.target.classList.contains('fa-xmark')) return;
            document.getElementById('stationModal').classList.add('hidden');
            document.body.style.overflow = 'auto';
        }

        // 초기 실행
        renderStations(stations);
    </script>
